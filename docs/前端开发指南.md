## API 调用封装

### 使用原生 fetch

```typescript
// frontend/src/api/query.ts

const API_BASE = '/api'

// 同步查询
export async function query(question: string): Promise<QueryResult> {
  const response = await fetch(`${API_BASE}/query`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ question })
  })

  if (!response.ok) {
    throw new Error(`Query failed: ${response.statusText}`)
  }

  return response.json()
}

// SSE 流式查询
export function streamQuery(
  question: string,
  onEvent: (event: SSEEvent) => void,
  onError?: (error: Error) => void
): () => void {
  const eventSource = new EventSource(
    `${API_BASE}/stream?question=${encodeURIComponent(question)}`
  )

  eventSource.onmessage = (e) => {
    try {
      const data = JSON.parse(e.data)
      onEvent({ event: data.event || 'message', data })
    } catch (err) {
      console.error('Failed to parse SSE data:', err)
    }
  }

  eventSource.onerror = (err) => {
    console.error('SSE error:', err)
    eventSource.close()
    onError?.(new Error('SSE connection failed'))
  }

  // 返回取消函数
  return () => eventSource.close()
}

// 获取图结构
export async function getGraphStructure(): Promise<GraphStructure> {
  const response = await fetch(`${API_BASE}/graph/structure`)
  return response.json()
}

// 获取历史记录
export async function getHistory(params?: {
  page?: number
  page_size?: number
  is_favorite?: boolean
  search?: string
}): Promise<{ items: HistoryItem[]; total: number }> {
  const searchParams = new URLSearchParams()
  if (params?.page) searchParams.set('page', params.page.toString())
  if (params?.page_size) searchParams.set('page_size', params.page_size.toString())
  if (params?.is_favorite !== undefined) searchParams.set('is_favorite', params.is_favorite.toString())
  if (params?.search) searchParams.set('search', params.search)

  const response = await fetch(`${API_BASE}/history?${searchParams}`)
  return response.json()
}

// 删除历史记录
export async function deleteHistory(id: number): Promise<void> {
  await fetch(`${API_BASE}/history/${id}`, { method: 'DELETE' })
}

// 切换收藏状态
export async function toggleFavorite(id: number, is_favorite: boolean): Promise<void> {
  await fetch(`${API_BASE}/history/${id}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ is_favorite })
  })
}
```

## Pinia 状态管理

### 定义 Store

```typescript
// frontend/src/stores/queryStore.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { QueryState, StepInfo, QueryResult } from '@/types'
import { streamQuery } from '@/api/query'

export const useQueryStore = defineStore('query', () => {
  // 状态
  const question = ref('')
  const status = ref<'idle' | 'running' | 'success' | 'error'>('idle')
  const currentStep = ref('')
  const steps = ref<StepInfo[]>([])
  const result = ref<QueryResult | null>(null)
  const error = ref<string | null>(null)

  // 计算属性
  const isRunning = computed(() => status.value === 'running')
  const hasResult = computed(() => result.value !== null)

  // 取消函数
  let cancelStream: (() => void) | null = null

  // 方法
  function startQuery(q: string) {
    question.value = q
    status.value = 'running'
    currentStep.value = ''
    steps.value = []
    result.value = null
    error.value = null

    cancelStream = streamQuery(q, handleEvent, handleError)
  }

  function handleEvent(event: SSEEvent) {
    switch (event.event) {
      case 'node_start':
        currentStep.value = event.data.node
        updateStep(event.data.node, 'running')
        break

      case 'node_complete':
        updateStep(event.data.node, 'completed', event.data.output)
        break

      case 'result':
        result.value = event.data
        status.value = 'success'
        cancelStream?.()
        break
    }
  }

  function handleError(err: Error) {
    status.value = 'error'
    error.value = err.message
  }

  function updateStep(nodeId: string, status: string, output?: any) {
    const existing = steps.value.find(s => s.id === nodeId)
    if (existing) {
      existing.status = status as any
      if (output) existing.output = output
    } else {
      steps.value.push({
        id: nodeId,
        label: nodeId,
        status: status as any,
        output
      })
    }
  }

  function cancel() {
    cancelStream?.()
    status.value = 'idle'
  }

  function reset() {
    question.value = ''
    status.value = 'idle'
    currentStep.value = ''
    steps.value = []
    result.value = null
    error.value = null
  }

  return {
    // 状态
    question,
    status,
    currentStep,
    steps,
    result,
    error,

    // 计算属性
    isRunning,
    hasResult,

    // 方法
    startQuery,
    cancel,
    reset
  }
})
```

## Vue 组件开发

### 组件模板

```vue
<!-- frontend/src/views/components/QueryInput.vue -->
<script setup lang="ts">
import { ref } from 'vue'
import { useQueryStore } from '@/stores/queryStore'

const store = useQueryStore()
const inputQuestion = ref('')

function submit() {
  if (inputQuestion.value.trim()) {
    store.startQuery(inputQuestion.value.trim())
  }
}
</script>

<template>
  <div class="flex gap-2">
    <input
      v-model="inputQuestion"
      type="text"
      placeholder="输入您的问题..."
      class="flex-1 px-4 py-2 border rounded-lg"
      @keyup.enter="submit"
      :disabled="store.isRunning"
    />
    <button
      @click="submit"
      :disabled="store.isRunning || !inputQuestion.trim()"
      class="px-6 py-2 bg-blue-500 text-white rounded-lg disabled:opacity-50"
    >
      查询
    </button>
  </div>
</template>
```

### 流程图组件

```vue
<!-- frontend/src/views/components/FlowGraph.vue -->
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { VueFlow, useVueFlow } from '@vue-flow/core'
import '@vue-flow/core/dist/style.css'
import { getGraphStructure } from '@/api/query'
import type { GraphNode, GraphEdge } from '@/types'

const nodes = ref<GraphNode[]>([])
const edges = ref<GraphEdge[]>([])

onMounted(async () => {
  const structure = await getGraphStructure()
  nodes.value = structure.nodes.map((n, i) => ({
    id: n.id,
    label: n.label,
    position: { x: 100 * i, y: 100 },
    type: n.type || 'default'
  }))
  edges.value = structure.edges.map(e => ({
    id: e.id,
    source: e.source,
    target: e.target,
    label: e.label
  }))
})
</script>

<template>
  <div class="h-64 border rounded-lg">
    <VueFlow :nodes="nodes" :edges="edges" />
  </div>
</template>
```

### 图表组件

```vue
<!-- frontend/src/views/components/ChartPanel.vue -->
<script setup lang="ts">
import { ref, watch, onMounted } from 'vue'
import * as echarts from 'echarts'
import type { QueryResult } from '@/types'

const props = defineProps<{
  result: QueryResult | null
}>()

const chartRef = ref<HTMLDivElement | null>(null)
let chart: echarts.ECharts | null = null

onMounted(() => {
  if (chartRef.value) {
    chart = echarts.init(chartRef.value)
  }
})

watch(() => props.result, (result) => {
  if (!result || !chart) return

  const option = {
    xAxis: {
      type: 'category',
      data: result.rows.map(r => r[0])
    },
    yAxis: {
      type: 'value'
    },
    series: [{
      data: result.rows.map(r => r[1]),
      type: 'bar'
    }]
  }

  chart.setOption(option)
}, { immediate: true })
</script>

<template>
  <div ref="chartRef" class="h-64 w-full"></div>
</template>
```

## SSE 流式消费

### 处理 SSE 事件

```typescript
// SSE 事件处理
function handleSSEEvent(event: SSEEvent) {
  switch (event.event) {
    case 'init':
      // 初始化图结构
      graphStructure.value = event.data.graph
      break

    case 'node_start':
      // 节点开始
      updateNodeStatus(event.data.node, 'running')
      break

    case 'node_complete':
      // 节点完成
      updateNodeStatus(event.data.node, 'completed')
      updateNodeOutput(event.data.node, event.data.output)
      break

    case 'result':
      // 最终结果
      result.value = event.data
      status.value = 'success'
      break

    case 'error':
      // 错误
      error.value = event.data.message
      status.value = 'error'
      break
  }
}
```

## 最佳实践

### ✅ 推荐做法

1. **组件设计**
   - 单一职责
   - Props 和 Emits 类型化
   - 使用 Composition API

2. **状态管理**
   - 集中管理查询状态
   - 使用 computed 派生状态
   - 异步操作在 Store 中处理

3. **API 调用**
   - 统一错误处理
   - 请求取消支持
   - TypeScript 类型完整

4. **性能优化**
   - 组件懒加载
   - 大列表虚拟滚动
   - 图表按需渲染

### ❌ 避免做法

1. **组件设计**
   - ❌ 过度嵌套组件
   - ❌ 缺少类型定义
   - ❌ 直接操作 DOM

2. **状态管理**
   - ❌ 分散的状态定义
   - ❌ 直接修改 props
   - ❌ 异步操作在组件中

3. **API 调用**
   - ❌ 硬编码 URL
   - ❌ 缺少错误处理
   - ❌ 不处理请求取消
